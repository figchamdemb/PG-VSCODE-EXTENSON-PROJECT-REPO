### 1.3 PostgreSQL High Availability Setup

```bash
# PRIMARY — /etc/postgresql/16/main/postgresql.conf
listen_addresses = '*'
wal_level = replica
max_wal_senders = 5
max_replication_slots = 5
hot_standby = on
synchronous_commit = on          # Never lose a transaction
synchronous_standby_names = '*'  # Wait for standby to confirm write

# PRIMARY — /etc/postgresql/16/main/pg_hba.conf
# Allow replication from standby private IP
host replication replicator 10.0.1.2/32 scram-sha-256

# STANDBY — clone from primary
pg_basebackup -h 10.0.1.1 -U replicator -D /var/lib/postgresql/16/main \
  -P -Xs -R --checkpoint=fast

# STANDBY — /etc/postgresql/16/main/postgresql.conf
hot_standby = on
primary_conninfo = 'host=10.0.1.1 port=5432 user=replicator'
```

Automatic failover using repmgr: repmgrd monitors the cluster and facilitates automatic failover — it constantly checks the health of the primary server and standby servers. If the primary goes down, repmgrd automatically elects a new primary from available standby servers.

```bash
# Install repmgr on both servers
apt install postgresql-16-repmgr -y

# /etc/repmgr.conf on PRIMARY
node_id=1
node_name=physicalotp-primary
conninfo='host=10.0.1.1 dbname=physicalotp user=repmgr'
data_directory='/var/lib/postgresql/16/main'
failover=automatic
promote_command='/usr/bin/repmgr standby promote -f /etc/repmgr.conf'
follow_command='/usr/bin/repmgr standby follow -f /etc/repmgr.conf'
```

### Complete PostgreSQL Schema

```sql
-- ============================================================
-- SCHEMA: physicalotp
-- All tables, indexes, constraints, and comments
-- ============================================================

CREATE EXTENSION IF NOT EXISTS "pgcrypto";  -- For gen_random_uuid()
CREATE EXTENSION IF NOT EXISTS "pg_stat_statements";  -- Query monitoring

-- ------------------------------------------------------------
-- TABLE: clients
-- Banks and fintechs that integrate our SDK
-- ------------------------------------------------------------
CREATE TABLE clients (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name                VARCHAR(200) NOT NULL,          -- "Barclays UK"
    api_key_id          VARCHAR(100) NOT NULL UNIQUE,   -- "pk_live_barclays_uk_..."
    api_key_hash        BYTEA NOT NULL,                 -- bcrypt hash of secret key
    environment         VARCHAR(20) NOT NULL            -- 'sandbox' | 'production'
                        CHECK (environment IN ('sandbox', 'production')),
    status              VARCHAR(20) NOT NULL DEFAULT 'active'
                        CHECK (status IN ('active', 'suspended', 'terminated')),
    plan                VARCHAR(20) NOT NULL DEFAULT 'growth'
                        CHECK (plan IN ('free', 'growth', 'scale', 'enterprise')),
    monthly_limit       BIGINT NOT NULL DEFAULT 1000,   -- verifications/month cap
    per_verify_price    NUMERIC(10,6) NOT NULL DEFAULT 0.008,  -- USD
    webhook_url         VARCHAR(500),                   -- where we POST results
    webhook_secret      BYTEA,                          -- HMAC secret for webhook
    binary_hash         VARCHAR(64),                    -- for on-premise: expected binary hash
    licence_expiry      TIMESTAMPTZ,                    -- NULL = cloud, date = on-premise
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_clients_api_key_id ON clients(api_key_id);
CREATE INDEX idx_clients_status ON clients(status);

-- ------------------------------------------------------------
-- TABLE: registered_devices
-- One row per registered user device — our core data
-- This is ALL we store about users. Nothing else.
-- ------------------------------------------------------------
CREATE TABLE registered_devices (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    client_id           UUID NOT NULL REFERENCES clients(id) ON DELETE CASCADE,
    
    -- The device identifier — random UUID generated on device
    -- We do not know who this belongs to — only the client does
    device_id           VARCHAR(36) NOT NULL,
    
    -- Ed25519 public key from device Secure Enclave / StrongBox
    -- 32 bytes — this is all we need to verify signatures
    public_key          BYTEA NOT NULL,
    
    -- Device attestation certificate
    -- Apple DeviceCheck / Google Play Integrity proof
    -- Proves this key was generated in real hardware (not emulator/rooted)
    attestation_cert    BYTEA,
    attestation_type    VARCHAR(20)  -- 'apple_devicecheck' | 'google_play_integrity'
                        CHECK (attestation_type IN ('apple_devicecheck', 'google_play_integrity', 'none')),
    
    -- ZKP commitment for recovery code
    -- SHA256(recovery_phrase + salt) — cannot be reversed
    -- Only the user's piece of paper has the actual phrase
    zkp_commitment      BYTEA,
    zkp_salt            BYTEA,  -- random salt used in commitment generation
    
    -- Device platform info (for analytics — no user data)
    platform            VARCHAR(20)  -- 'ios' | 'android'
                        CHECK (platform IN ('ios', 'android')),
    sdk_version         VARCHAR(20),  -- which version of our SDK they have
    
    -- Security state
    status              VARCHAR(20) NOT NULL DEFAULT 'active'
                        CHECK (status IN ('active', 'revoked', 'suspended')),
    revoked_at          TIMESTAMPTZ,
    revoked_reason      VARCHAR(100),
    
    -- Timestamps
    registered_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    last_verified_at    TIMESTAMPTZ,
    
    -- Uniqueness: one device_id per client (different clients = different namespaces)
    CONSTRAINT uq_client_device UNIQUE (client_id, device_id)
);

-- Indexes for fast lookup
CREATE INDEX idx_devices_client_device ON registered_devices(client_id, device_id);
CREATE INDEX idx_devices_status ON registered_devices(status);
CREATE INDEX idx_devices_last_verified ON registered_devices(last_verified_at);

-- ------------------------------------------------------------
-- TABLE: challenges
-- One-time challenges issued for verification
-- Short TTL — cleaned up automatically
-- ------------------------------------------------------------
-- NOTE: Challenges also live in Redis (30s TTL, auto-expire)
-- This table is the audit log only — for dispute resolution
CREATE TABLE challenges (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    client_id           UUID NOT NULL REFERENCES clients(id),
    device_id           VARCHAR(36) NOT NULL,
    
    -- The random challenge bytes (32 bytes, hex-encoded)
    challenge_bytes     BYTEA NOT NULL,
    
    -- Geolocation at time of challenge issue
    -- GPS coordinates from device hardware (not IP-based)
    request_lat         NUMERIC(10,7),  -- e.g. 51.5074000
    request_lon         NUMERIC(10,7),  -- e.g. -0.1278000
    request_geohash     VARCHAR(10),    -- GeoHash precision 5 (~5km box)
    
    -- TLS session fingerprint (for session binding)
    session_fingerprint VARCHAR(64),    -- SHA256 of TLS session ID + client IP
    
    -- State
    status              VARCHAR(20) NOT NULL DEFAULT 'pending'
                        CHECK (status IN ('pending', 'consumed', 'expired')),
    issued_at           TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    expires_at          TIMESTAMPTZ NOT NULL DEFAULT NOW() + INTERVAL '30 seconds',
    consumed_at         TIMESTAMPTZ
);

-- Auto-delete old challenges (keep 30 days for audit)
CREATE INDEX idx_challenges_expires ON challenges(expires_at);
CREATE INDEX idx_challenges_client_device ON challenges(client_id, device_id);

-- Cleanup job: DELETE FROM challenges WHERE expires_at < NOW() - INTERVAL '30 days';

-- ------------------------------------------------------------
-- TABLE: verifications
-- Audit log of every verification attempt — pass or fail
-- This is the most important table for compliance
-- ------------------------------------------------------------
CREATE TABLE verifications (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    client_id           UUID NOT NULL REFERENCES clients(id),
    device_id           VARCHAR(36) NOT NULL,
    challenge_id        UUID REFERENCES challenges(id),
    
    -- Result
    result              VARCHAR(20) NOT NULL
                        CHECK (result IN ('verified', 'failed', 'expired', 'spoofed')),
    
    -- Which criteria failed (for failed verifications)
    -- Stored as array of strings: ['device_signature', 'session_binding', 'gps_location']
    failed_criteria     TEXT[],
    
    -- Risk signals detected
    risk_score          INTEGER DEFAULT 0,  -- 0-100
    risk_flags          TEXT[],             -- ['gps_ip_mismatch', 'velocity_anomaly']
    
    -- Redemption context (where the verification was submitted from)
    -- NOTE: We store GPS coords for fraud detection — NOT for profiling
    -- Retained for 90 days then deleted
    redeem_lat          NUMERIC(10,7),
    redeem_lon          NUMERIC(10,7),
    redeem_geohash      VARCHAR(10),
    
    -- Timestamps
    verified_at         TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Indexes for billing queries and fraud detection
CREATE INDEX idx_verifications_client ON verifications(client_id, verified_at);
CREATE INDEX idx_verifications_device ON verifications(device_id, verified_at);
CREATE INDEX idx_verifications_result ON verifications(result, verified_at);

-- Partitioned by month for performance at scale
-- At 10M verifications/month this table grows fast
-- Partition by range on verified_at

-- ------------------------------------------------------------
-- TABLE: billing_usage
-- Monthly rollup of verifications per client — for invoicing
-- ------------------------------------------------------------
CREATE TABLE billing_usage (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    client_id           UUID NOT NULL REFERENCES clients(id),
    billing_month       DATE NOT NULL,  -- First day of month: 2026-02-01
    
    total_verifications BIGINT NOT NULL DEFAULT 0,
    successful          BIGINT NOT NULL DEFAULT 0,
    failed              BIGINT NOT NULL DEFAULT 0,
    
    amount_usd          NUMERIC(10,2),  -- Calculated at month end
    invoice_status      VARCHAR(20) DEFAULT 'pending'
                        CHECK (invoice_status IN ('pending', 'invoiced', 'paid', 'overdue')),
    invoice_id          VARCHAR(100),   -- Stripe invoice ID
    
    calculated_at       TIMESTAMPTZ,
    
    CONSTRAINT uq_billing_client_month UNIQUE (client_id, billing_month)
);

-- ------------------------------------------------------------
-- TABLE: licence_heartbeats  
-- On-premise client agent check-ins
-- ------------------------------------------------------------
CREATE TABLE licence_heartbeats (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    client_id           UUID NOT NULL REFERENCES clients(id),
    
    -- The binary hash proves they haven't tampered with our client agent
    reported_binary_hash    VARCHAR(64) NOT NULL,
    expected_binary_hash    VARCHAR(64) NOT NULL,
    binary_valid            BOOLEAN NOT NULL,
    
    -- Volume reported by the client agent (for billing)
    verifications_since_last BIGINT NOT NULL DEFAULT 0,
    
    -- Operating certificate we issue (valid 24 hours)
    certificate_issued  BYTEA,          -- signed operating certificate
    certificate_expiry  TIMESTAMPTZ,
    
    heartbeat_at        TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    client_ip           INET            -- their server IP (for on-premise)
);

CREATE INDEX idx_heartbeats_client ON licence_heartbeats(client_id, heartbeat_at);

-- ------------------------------------------------------------
-- TABLE: admin_users
-- Our own staff who access the admin panel
-- Protected by the same PhysicalOTP system we sell
-- ------------------------------------------------------------
CREATE TABLE admin_users (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email               VARCHAR(200) NOT NULL UNIQUE,
    email_hash          BYTEA NOT NULL,     -- For PhysicalOTP email verification
    role                VARCHAR(20) NOT NULL DEFAULT 'support'
                        CHECK (role IN ('superadmin', 'admin', 'support', 'billing')),
    device_id           VARCHAR(36),        -- Their registered device (PhysicalOTP)
    public_key          BYTEA,              -- Their Secure Enclave key
    status              VARCHAR(20) NOT NULL DEFAULT 'active',
    last_login_at       TIMESTAMPTZ,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ------------------------------------------------------------
-- VIEWS — for admin panel queries
-- ------------------------------------------------------------

-- Monthly usage summary per client
CREATE VIEW v_client_monthly_usage AS
SELECT 
    c.id AS client_id,
    c.name AS client_name,
    c.plan,
    DATE_TRUNC('month', v.verified_at) AS month,
    COUNT(*) AS total_verifications,
    SUM(CASE WHEN v.result = 'verified' THEN 1 ELSE 0 END) AS successful,
    SUM(CASE WHEN v.result = 'failed' THEN 1 ELSE 0 END) AS failed,
    SUM(CASE WHEN v.result = 'spoofed' THEN 1 ELSE 0 END) AS spoofed,
    ROUND(AVG(v.risk_score), 2) AS avg_risk_score
FROM verifications v
JOIN clients c ON v.client_id = c.id
GROUP BY c.id, c.name, c.plan, DATE_TRUNC('month', v.verified_at);

-- Device health summary (no user data — just counts)
CREATE VIEW v_device_health AS
SELECT
    client_id,
    platform,
    sdk_version,
    COUNT(*) AS registered_devices,
    COUNT(CASE WHEN status = 'active' THEN 1 END) AS active_devices,
    COUNT(CASE WHEN last_verified_at > NOW() - INTERVAL '7 days' THEN 1 END) AS active_7d
FROM registered_devices
GROUP BY client_id, platform, sdk_version;
```

### Key Database Rules

```
INDEXES — what we index and why:
  registered_devices(client_id, device_id)  → primary lookup on every verify
  verifications(client_id, verified_at)     → billing queries
  verifications(device_id, verified_at)     → fraud detection (velocity check)
  challenges(expires_at)                    → cleanup job performance

RETENTION POLICY:
  challenges:            30 days (audit), then DELETE
  verifications:         90 days (redeem_lat/lon), then NULL those columns
                         Keep the row (result + timestamps) for 7 years (compliance)
  billing_usage:         Forever (financial records)
  licence_heartbeats:    90 days, then DELETE

WHAT WE NEVER STORE:
  User names, emails, phone numbers
  OTP values (not even hashed)
  Biometric data
  Session content
  Transaction amounts or details
  Any data that could identify a real person without the client's mapping
```

### Core API Endpoints

```python
# main.py
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from contextlib import asynccontextmanager
import uvicorn

@asynccontextmanager
async def lifespan(app: FastAPI):
    # Startup: connect DB, Redis, load HSM key reference
    await database.connect()
    await redis_client.connect()
    await crypto.load_root_key()
    yield
    # Shutdown: clean up connections
    await database.disconnect()

app = FastAPI(
    title="PhysicalOTP API",
    version="1.0.0",
    docs_url=None,      # No public Swagger docs — security
    redoc_url=None,
    lifespan=lifespan
)

# ============================================================
# POST /api/v1/register
# Register a new device — store UUID + public key
# ============================================================
@app.post("/api/v1/register")
async def register_device(
    request: RegisterRequest,
    client: Client = Depends(verify_api_key)
):
    """
    Called when user installs the app for the first time.
    Stores their Secure Enclave public key.
    That is all. We know nothing else about this user.
    """
    # Verify device attestation (proves real hardware, not emulator)
    attestation_valid = await attestation.verify(
        platform=request.platform,
        attestation_cert=request.attestation_cert,
        device_id=request.device_id,
        challenge=request.attestation_challenge
    )
    if not attestation_valid:
        raise HTTPException(400, "Device attestation failed")
    
    # Store device — 2 rows in DB: registered_devices
    device = await db.register_device(
        client_id=client.id,
        device_id=request.device_id,
        public_key=bytes.fromhex(request.public_key),
        attestation_cert=request.attestation_cert,
        platform=request.platform
    )
    
    return {"status": "registered", "device_id": device.device_id}


# ============================================================
# GET /api/v1/challenge
# Issue a fresh challenge for this device
# ============================================================
@app.get("/api/v1/challenge")
async def get_challenge(
    device_id: str,
    lat: float,
    lon: float,
    client: Client = Depends(verify_api_key),
    request: Request = None
):
    """
    Returns a random 32-byte challenge.
    Challenge is stored in Redis with 30s TTL.
    Bound to: device_id + GPS location + TLS session fingerprint.
    """
    import secrets
    
    challenge_bytes = secrets.token_bytes(32)
    challenge_id = str(uuid4())
    
    # Compute session fingerprint from TLS session + client IP
    session_fp = crypto.fingerprint_session(
        request.headers.get("X-TLS-Session-ID", ""),
        request.client.host
    )
    
    # Compute GeoHash from GPS (precision 5 = ~5km box)
    geohash = geolocation.encode(lat, lon, precision=5)
    
    # Store in Redis (30s TTL — auto-expires)
    await redis.setex(
        f"challenge:{challenge_id}",
        30,
        json.dumps({
            "challenge_bytes": challenge_bytes.hex(),
            "client_id": str(client.id),
            "device_id": device_id,
            "geohash": geohash,
            "session_fp": session_fp,
            "issued_at": time.time()
        })
    )
    
    return {
        "challenge_id": challenge_id,
        "challenge_bytes": challenge_bytes.hex(),
        "expires_in": 30
    }


# ============================================================
# POST /api/v1/verify
# Verify a signed challenge — the core endpoint
# All 5 criteria checked here
# ============================================================
@app.post("/api/v1/verify")
async def verify(
    request: VerifyRequest,
    client: Client = Depends(verify_api_key)
):
    """
    The core verification endpoint.
    Returns: verified | failed | expired | spoofed
    Banks call this to confirm physical presence.
    """
    # ATOMIC: get and delete challenge from Redis in one operation
    # First use wins — prevents replay
    raw = await redis.getdel(f"challenge:{request.challenge_id}")
    if not raw:
        return VerifyResponse(result="expired", failed_criteria=["time_window"])
    
    stored = json.loads(raw)
    failed = []
    risk_flags = []
    
    # CRITERION 1 — Device signature (Secure Enclave hardware proof)
    device = await db.get_device(client.id, request.device_id)
    if not device:
        return VerifyResponse(result="failed", failed_criteria=["device_not_registered"])
    
    sig_valid = crypto.verify_ed25519(
        public_key=device.public_key,
        message=bytes.fromhex(stored["challenge_bytes"]),
        signature=bytes.fromhex(request.device_signature)
    )
    if not sig_valid:
        failed.append("device_signature")
    
    # CRITERION 2 — Session binding
    redeem_session_fp = crypto.fingerprint_session(
        request.tls_session_id or "",
        request.client_ip or ""
    )
    if redeem_session_fp != stored["session_fp"]:
        failed.append("session_binding")
    
    # CRITERION 3 — GPS location binding
    redeem_geohash = geolocation.encode(request.lat, request.lon, precision=5)
    if not geolocation.within_tolerance(stored["geohash"], redeem_geohash, km=10):
        failed.append("gps_location")
        risk_flags.append("geolocation_mismatch")
    
    # CRITERION 4 — Time window (enforced by Redis TTL — already handled above)
    # CRITERION 5 — One-time use (enforced by GETDEL — already handled above)
    
    # Risk scoring
    risk_score, risk_flags = await risk.calculate(
        device_id=request.device_id,
        client_id=client.id,
        gps_lat=request.lat,
        gps_lon=request.lon,
        client_ip=request.client_ip,
        failed_criteria=failed,
        existing_flags=risk_flags
    )
    
    result = "verified" if not failed else ("spoofed" if "device_signature" in failed else "failed")
    
    # Log to DB (async — don't block the response)
    asyncio.create_task(db.log_verification(
        client_id=client.id,
        device_id=request.device_id,
        challenge_id=request.challenge_id,
        result=result,
        failed_criteria=failed,
        risk_score=risk_score,
        risk_flags=risk_flags,
        redeem_lat=request.lat,
        redeem_lon=request.lon
    ))
    
    # Update last_verified_at
    asyncio.create_task(db.update_device_last_verified(client.id, request.device_id))
    
    # Fire webhook to client (async)
    if client.webhook_url and result == "verified":
        asyncio.create_task(webhook.send(client, {
            "device_id": request.device_id,
            "result": result,
            "verified_at": datetime.utcnow().isoformat()
        }))
    
    return VerifyResponse(
        result=result,
        failed_criteria=failed if failed else [],
        risk_score=risk_score
    )
```

### What Happens When Hetzner Goes Down

```
SCENARIO: Primary Hetzner server fails completely

T+0s    Primary server stops responding
T+5s    Keepalived health check fails (checks every 5s)
T+10s   Keepalived on standby detects: primary is down
T+12s   Keepalived triggers: promote_floating_ip.sh
T+15s   Hetzner API: Floating IP reassigned to standby server
T+20s   Cloudflare DNS already pointing to Floating IP (unchanged)
T+20s   Traffic flows to standby server automatically
T+25s   PostgreSQL replica promoted to primary (repmgr)
T+30s   Full service restored — 30 seconds maximum downtime

WHAT BANKS EXPERIENCE:
  30 seconds of failed verifications
  Our SDK handles this gracefully:
    - Retry logic: 3 retries with exponential backoff
    - If all retries fail: SDK returns PHYSICALOTP_SERVICE_UNAVAILABLE
    - Bank's app shows: "Verification temporarily unavailable"
    - Bank can fall back to: their existing OTP (SMS/TOTP) with lower security
    - We send: status page update + webhook notification "service degraded"

BANK FALLBACK CONFIGURATION (they set this in our dashboard):
  fallback_mode: "legacy_otp"   # Use SMS OTP if we are down
               | "block"        # Block the action entirely until we recover
               | "allow"        # Allow without verification (not recommended)
  
  Most banks set: fallback_mode = "legacy_otp"
  This means: users are never completely locked out
              but lose the physical verification protection
              during the ~30 second failover window
```

```
  Banks using Flutter: import 'package:physicalotp_flutter/physicalotp_flutter.dart'
```

```swift
// Sources/PhysicalOTP/PhysicalOTPClient.swift

import Foundation
import LocalAuthentication
import Security
import CryptoKit

public class PhysicalOTP {
    
    // MARK: — Configuration
    
    private static var apiKey: String = ""
    private static var apiBaseURL: String = "https://api.physicalotp.com/api/v1"
    private static var environment: Environment = .production
    
    public enum Environment { case sandbox, production }
    
    public static func configure(apiKey: String, environment: Environment = .production) {
        Self.apiKey = apiKey
        Self.environment = environment
        if environment == .sandbox {
            Self.apiBaseURL = "https://sandbox.physicalotp.com/api/v1"
        }
    }
    
    // MARK: — Device Registration
    
    public static func registerDevice() async -> Result<String, PhysicalOTPError> {
        // 1. Generate key pair in Secure Enclave
        //    The private key NEVER leaves the Secure Enclave chip
        guard let key = try? generateSecureEnclaveKey() else {
            return .failure(.hardwareNotSupported)
        }
        
        // 2. Extract public key (safe to share)
        let publicKey = SecureEnclave.P256.Signing.PublicKey(key).rawRepresentation
        
        // 3. Generate device ID (random UUID — not linked to anything)
        let deviceID = UUID().uuidString
        KeychainHelper.save(deviceID, forKey: "physicalotp_device_id")
        
        // 4. Get device attestation (proves this is real hardware, not emulator)
        let attestation = try? await getAppleDeviceCheckToken()
        
        // 5. Register with our server
        let result = await APIClient.register(
            deviceID: deviceID,
            publicKey: publicKey.hexString,
            attestation: attestation,
            platform: "ios"
        )
        
        switch result {
        case .success:
            return .success(deviceID)
        case .failure(let error):
            return .failure(error)
        }
    }
    
    // MARK: — Physical Presence Verification (THE CORE FUNCTION)
    
    public static func verify(reason: String) async -> VerificationResult {
        guard let deviceID = KeychainHelper.load(forKey: "physicalotp_device_id") else {
            return .deviceNotRegistered
        }
        
        // 1. Get current GPS location (hardware GPS — VPN cannot spoof this)
        guard let location = await LocationManager.getCurrentLocation() else {
            return .locationUnavailable
        }
        
        // 2. Request challenge from server (bound to this session + location)
        let challengeResult = await APIClient.getChallenge(
            deviceID: deviceID,
            lat: location.coordinate.latitude,
            lon: location.coordinate.longitude
        )
        guard case .success(let challenge) = challengeResult else {
            return .networkError
        }
        
        // 3. Biometric gate — REAL hardware biometric required
        //    .biometryCurrentSet means: only currently enrolled face/fingerprint
        //    Adding a new face to Face ID invalidates the key — security feature
        let context = LAContext()
        context.localizedReason = reason
        
        do {
            try await context.evaluatePolicy(
                .deviceOwnerAuthenticationWithBiometricsAndWatch,
                localizedReason: reason
            )
        } catch {
            return .biometricFailed
        }
        
        // 4. Sign the challenge bytes using Secure Enclave private key
        //    This step requires biometric unlock — it just happened above
        //    The signature PROVES this specific hardware chip was present
        guard let signature = try? signWithSecureEnclave(
            data: Data(hex: challenge.challengeBytes)
        ) else {
            return .signingFailed
        }
        
        // 5. Submit to server with all binding material
        let verifyResult = await APIClient.verify(
            challengeID: challenge.challengeID,
            deviceID: deviceID,
            deviceSignature: signature.hexString,
            lat: location.coordinate.latitude,
            lon: location.coordinate.longitude
        )
        
        switch verifyResult {
        case .success(let response):
            return response.result == "verified" ? .verified : .failed(response.failedCriteria)
        case .failure:
            return .networkError
        }
    }
    
    // MARK: — Secure Enclave Helpers
    
    private static func generateSecureEnclaveKey() throws -> SecureEnclave.P256.Signing.PrivateKey {
        return try SecureEnclave.P256.Signing.PrivateKey(
            accessControl: SecAccessControlCreateWithFlags(
                nil,
                kSecAttrAccessibleWhenUnlockedThisDeviceOnly,
                [.privateKeyUsage, .biometryCurrentSet],  // Biometric required to use key
                nil
            )!
        )
    }
    
    private static func signWithSecureEnclave(data: Data) throws -> Data {
        let key = try SecureEnclave.P256.Signing.PrivateKey(
            dataRepresentation: KeychainHelper.loadKey(forKey: "physicalotp_private_key")!
        )
        let signature = try key.signature(for: data)
        return signature.rawRepresentation
    }
}

// MARK: — Result Types

public enum VerificationResult {
    case verified
    case failed([String])           // List of failed criteria
    case biometricFailed
    case deviceNotRegistered
    case locationUnavailable
    case signingFailed
    case networkError
    case deviceIntegrityFailed      // Jailbroken device detected
}
```

### Android SDK (Kotlin — Core Native)

```kotlin
// PhysicalOTP/src/main/kotlin/com/physicalotp/PhysicalOTPClient.kt

package com.physicalotp

import android.content.Context
import android.security.keystore.KeyGenParameterSpec
import android.security.keystore.KeyProperties
import androidx.biometric.BiometricPrompt
import androidx.fragment.app.FragmentActivity
import java.security.KeyPairGenerator
import java.security.KeyStore

class PhysicalOTP(private val context: Context) {

    companion object {
        private var apiKey: String = ""
        private var apiBaseURL: String = "https://api.physicalotp.com/api/v1"
        
        fun configure(context: Context, apiKey: String, sandbox: Boolean = false) {
            this.apiKey = apiKey
            if (sandbox) apiBaseURL = "https://sandbox.physicalotp.com/api/v1"
        }
    }
    
    // KEY ALIAS in Android Keystore — private key never leaves hardware
    private val KEY_ALIAS = "physicalotp_signing_key"
    
    fun registerDevice(activity: FragmentActivity, callback: (Result<String>) -> Unit) {
        // 1. Generate key pair in Android Keystore (StrongBox if available)
        //    StrongBox = dedicated security chip (Pixel 3+, Samsung S9+)
        //    Falls back to TEE (Trusted Execution Environment) if no StrongBox
        val keySpec = KeyGenParameterSpec.Builder(
            KEY_ALIAS,
            KeyProperties.PURPOSE_SIGN
        )
            .setAlgorithmParameterSpec(java.security.spec.ECGenParameterSpec("secp256r1"))
            .setDigests(KeyProperties.DIGEST_SHA256)
            .setUserAuthenticationRequired(true)        // Biometric required to use key
            .setUserAuthenticationParameters(0, KeyProperties.AUTH_BIOMETRIC_STRONG)
            .setInvalidatedByBiometricEnrollment(true)  // New fingerprint = new key
            .setIsStrongBoxBacked(isStrongBoxAvailable())  // Use StrongBox if available
            .build()
        
        val keyPairGen = KeyPairGenerator.getInstance(
            KeyProperties.KEY_ALGORITHM_EC, "AndroidKeyStore"
        )
        keyPairGen.initialize(keySpec)
        val keyPair = keyPairGen.generateKeyPair()
        
        // 2. Extract public key
        val publicKey = keyPair.public.encoded.toHexString()
        
        // 3. Device ID
        val deviceID = java.util.UUID.randomUUID().toString()
        PrefsHelper.save(context, "device_id", deviceID)
        
        // 4. Register with server
        ApiClient.register(deviceID, publicKey, "android") { result ->
            callback(result.map { deviceID })
        }
    }
    
    fun verify(
        activity: FragmentActivity,
        reason: String,
        callback: (VerificationResult) -> Unit
    ) {
        val deviceID = PrefsHelper.load(context, "device_id") ?: run {
            callback(VerificationResult.DeviceNotRegistered)
            return
        }
        
        // 1. Get GPS location (hardware GPS)
        LocationHelper.getCurrentLocation(context) { location ->
            if (location == null) {
                callback(VerificationResult.LocationUnavailable)
                return@getCurrentLocation
            }
            
            // 2. Get challenge from server
            ApiClient.getChallenge(deviceID, location.latitude, location.longitude) { challengeResult ->
                val challenge = challengeResult.getOrElse {
                    callback(VerificationResult.NetworkError)
                    return@getChallenge
                }
                
                // 3. Biometric prompt — triggers hardware biometric gate
                val biometricPrompt = BiometricPrompt(
                    activity,
                    activity.mainExecutor,
                    object : BiometricPrompt.AuthenticationCallback() {
                        override fun onAuthenticationSucceeded(result: BiometricPrompt.AuthenticationResult) {
                            // 4. Sign challenge with Secure hardware key
                            //    result.cryptoObject.signature is already initialized
                            //    with the hardware-backed private key
                            try {
                                val signature = result.cryptoObject?.signature!!
                                signature.update(challenge.challengeBytes.hexToByteArray())
                                val signedBytes = signature.sign()
                                
                                // 5. Submit to server
                                ApiClient.verify(
                                    challengeID = challenge.challengeID,
                                    deviceID = deviceID,
                                    signature = signedBytes.toHexString(),
                                    lat = location.latitude,
                                    lon = location.longitude
                                ) { verifyResult ->
                                    val response = verifyResult.getOrElse {
                                        callback(VerificationResult.NetworkError)
                                        return@verify
                                    }
                                    callback(
                                        if (response.result == "verified") VerificationResult.Verified
                                        else VerificationResult.Failed(response.failedCriteria)
                                    )
                                }
                            } catch (e: Exception) {
                                callback(VerificationResult.SigningFailed)
                            }
                        }
                        
                        override fun onAuthenticationFailed() {
                            callback(VerificationResult.BiometricFailed)
                        }
                    }
                )
                
                // Initialise signature with hardware key BEFORE showing prompt
                val keyStore = KeyStore.getInstance("AndroidKeyStore").apply { load(null) }
                val privateKey = keyStore.getKey(KEY_ALIAS, null)
                val signature = java.security.Signature.getInstance("SHA256withECDSA")
                signature.initSign(privateKey as java.security.PrivateKey)
                val cryptoObject = BiometricPrompt.CryptoObject(signature)
                
                val promptInfo = BiometricPrompt.PromptInfo.Builder()
                    .setTitle("Verify your identity")
                    .setSubtitle(reason)
                    .setNegativeButtonText("Cancel")
                    .setAllowedAuthenticators(BiometricManager.Authenticators.BIOMETRIC_STRONG)
                    .build()
                
                biometricPrompt.authenticate(promptInfo, cryptoObject)
            }
        }
    }
    
    private fun isStrongBoxAvailable(): Boolean {
        return context.packageManager.hasSystemFeature(
            android.content.pm.PackageManager.FEATURE_STRONGBOX_KEYSTORE
        )
    }
}

sealed class VerificationResult {
    object Verified : VerificationResult()
    data class Failed(val criteria: List<String>) : VerificationResult()
    object BiometricFailed : VerificationResult()
    object DeviceNotRegistered : VerificationResult()
    object LocationUnavailable : VerificationResult()
    object SigningFailed : VerificationResult()
    object NetworkError : VerificationResult()
    object DeviceIntegrityFailed : VerificationResult()
}
```

```typescript
// content/gmail.ts
// Intercepts the Gmail send button before email is sent

class GmailInterceptor {
  
  private protectedEmails: Set<string> = new Set();
  
  init() {
    // Load protected emails from storage
    chrome.storage.sync.get(['protected_emails'], (result) => {
      this.protectedEmails = new Set(result.protected_emails || []);
    });
    
    // Watch for Gmail's send button with MutationObserver
    // Gmail is a SPA — elements appear dynamically
    const observer = new MutationObserver(() => {
      this.attachSendInterceptors();
    });
    
    observer.observe(document.body, { childList: true, subtree: true });
    this.attachSendInterceptors();
  }
  
  private attachSendInterceptors() {
    // Gmail send button selector (updates may be needed as Gmail changes)
    const sendButtons = document.querySelectorAll(
      '[data-tooltip="Send ‪(Ctrl-Enter)‬"], [aria-label^="Send"]'
    );
    
    sendButtons.forEach(button => {
      if (button.getAttribute('data-physicalotp-intercepted')) return;
      button.setAttribute('data-physicalotp-intercepted', 'true');
      
      button.addEventListener('click', (e) => this.handleSendClick(e), true);
    });
  }
  
  private async handleSendClick(e: Event) {
    // Get current active Gmail account
    const fromEmail = this.getActiveGmailAccount();
    
    // Check if this email is in the protected list
    if (!this.protectedEmails.has(fromEmail)) {
      return; // Not protected — allow send normally
    }
    
    // INTERCEPT — prevent default send
    e.preventDefault();
    e.stopImmediatePropagation();
    
    // Get email content for signing
    const emailContent = this.getEmailContent();
    
    // Show verification overlay
    this.showVerificationOverlay(emailContent, fromEmail);
  }
  
  private showVerificationOverlay(content: EmailContent, fromEmail: string) {
    const overlay = document.createElement('div');
    overlay.id = 'physicalotp-overlay';
    overlay.innerHTML = `
      <div class="physicalotp-modal">
        <div class="physicalotp-icon">📱</div>
        <h3>Tap your phone to send</h3>
        <p>Sending from: <strong>${fromEmail}</strong></p>
        <p>To: ${content.to}</p>
        <p>Subject: ${content.subject}</p>
        <div class="physicalotp-spinner" id="physicalotp-status">
          Waiting for phone confirmation...
        </div>
        <button id="physicalotp-cancel">Cancel</button>
      </div>
    `;
    document.body.appendChild(overlay);
    
    // Cancel button
    overlay.querySelector('#physicalotp-cancel')!
      .addEventListener('click', () => overlay.remove());
    
    // Request phone tap verification
    this.requestPhoneTap(content, fromEmail, overlay);
  }
  
  private async requestPhoneTap(content: EmailContent, fromEmail: string, overlay: HTMLElement) {
    // Compute email hash (what we are signing)
    const emailHash = await crypto.sha256(
      `${content.to}|${content.subject}|${content.body}|${Date.now()}`
    );
    
    // Request verification from our server (which pushes to phone)
    const session = await APIClient.requestEmailSign({
      deviceID: storage.getDeviceID(),
      emailHash,
      fromEmail,
      preview: {
        to: content.to,
        subject: content.subject,
        bodyPreview: content.body.substring(0, 200)
      }
    });
    
    // Poll for phone approval (phone signs → sends to server → we poll)
    const result = await this.pollForApproval(session.sessionID);
    overlay.remove();
    
    if (result.approved) {
      // Phone approved — add signature header and send
      await this.addSignatureAndSend(content, result.signature);
    }
    // If denied or timeout: do nothing — email stays unsent
  }
  
  private async addSignatureAndSend(content: EmailContent, signature: string) {
    // Gmail doesn't allow adding custom headers via DOM
    // So we route via our background service worker which uses Gmail API
    chrome.runtime.sendMessage({
      type: 'SEND_SIGNED_EMAIL',
      content,
      signature
    });
  }
  
  private getActiveGmailAccount(): string {
    // Gmail shows current account in the URL: mail.google.com/mail/u/0/
    // Index 0, 1, 2 = different accounts
    const match = window.location.href.match(/\/u\/(\d+)\//);
    const accountIndex = match ? parseInt(match[1]) : 0;
    
    // Get email from page meta
    const emailMeta = document.querySelector(`[data-email]`);
    return emailMeta?.getAttribute('data-email') || '';
  }
}
```

```typescript
'use client';
import { useState } from 'react';

export default function AdminLogin() {
  const [status, setStatus] = useState<'idle' | 'waiting' | 'verified'>('idle');
  
  async function handleLogin(email: string) {
    setStatus('waiting');
    
    // Request a challenge for this admin's registered device
    const challenge = await fetch('/api/admin/challenge', {
      method: 'POST',
      body: JSON.stringify({ email })
    }).then(r => r.json());
    
    // Poll for phone tap completion
    // Admin sees: "Tap your phone to log in to the admin panel"
    const result = await pollForAdminVerification(challenge.sessionID);
    
    if (result.verified) {
      // Set session cookie, redirect to dashboard
      setStatus('verified');
      window.location.href = '/dashboard';
    }
  }
  
  return (
    <div className="login-screen">
      <h1>PhysicalOTP Admin</h1>
      {status === 'idle' && (
        <form onSubmit={(e) => {
          e.preventDefault();
          handleLogin(e.currentTarget.email.value);
        }}>
          <input name="email" type="email" placeholder="Admin email" />
          <button type="submit">Continue</button>
        </form>
      )}
      {status === 'waiting' && (
        <div className="waiting">
          <div className="phone-icon">📱</div>
          <p>Tap your registered phone to log in</p>
          <p className="hint">Push notification sent to your device</p>
        </div>
      )}
    </div>
  );
}
```

### Backup Strategy

```bash
# Daily backup script — runs at 02:00 UTC via cron

#!/bin/bash
DATE=$(date +%Y%m%d)
BACKUP_FILE="physicalotp-db-$DATE.sql.gz"

# 1. Dump PostgreSQL (from standby to avoid impact on primary)
pg_dump -h 10.0.1.2 -U backup_user physicalotp | gzip > /tmp/$BACKUP_FILE

# 2. Encrypt with AWS KMS key before upload
aws kms generate-data-key --key-id alias/physicalotp-backup \
  --key-spec AES_256 --output json > /tmp/data-key.json

# Encrypt the backup
openssl enc -aes-256-cbc -in /tmp/$BACKUP_FILE \
  -out /tmp/$BACKUP_FILE.enc \
  -K $(cat /tmp/data-key.json | jq -r .Plaintext) \
  -iv $(openssl rand -hex 16)

# 3. Upload to Cloudflare R2 (zero egress cost on restore)
aws s3 cp /tmp/$BACKUP_FILE.enc \
  s3://physicalotp-backups/db/$BACKUP_FILE.enc \
  --endpoint-url https://accountid.r2.cloudflarestorage.com

# 4. Clean up local files
rm /tmp/$BACKUP_FILE /tmp/$BACKUP_FILE.enc /tmp/data-key.json

# 5. Verify backup was uploaded
aws s3 ls s3://physicalotp-backups/db/$BACKUP_FILE.enc \
  --endpoint-url https://accountid.r2.cloudflarestorage.com || \
  echo "BACKUP FAILED" | mail -s "PhysicalOTP Backup Failed" ops@physicalotp.com

# Delete backups older than 30 days
aws s3 ls s3://physicalotp-backups/db/ \
  --endpoint-url https://accountid.r2.cloudflarestorage.com | \
  awk '{print $4}' | while read file; do
    # Delete if older than 30 days
    ...
  done
```
